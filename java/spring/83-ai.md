```java
// ─── Basic ChatClient ─────────────────────────────────────
@Service
public class AiChatService {

    private final ChatClient chatClient;

    // Auto-configured ChatClient.Builder inject olunur
    public AiChatService(ChatClient.Builder builder) {
        this.chatClient = builder.build();
    }

    // Sadə sual-cavab
    public String ask(String question) {
        return chatClient.prompt()
            .user(question)
            .call()
            .content();
    }

    // System prompt + user message
    public String askWithContext(String systemPrompt, String userMessage) {
        return chatClient.prompt()
            .system(systemPrompt)
            .user(userMessage)
            .call()
            .content();
    }
}

// ─── Streaming response ───────────────────────────────────
@RestController
@RequestMapping("/api/ai")
public class AiController {

    private final ChatClient chatClient;

    // Streaming — token-by-token (Server-Sent Events)
    @GetMapping(value = "/stream", produces = MediaType.TEXT_EVENT_STREAM_VALUE)
    public Flux<String> streamResponse(@RequestParam String message) {
        return chatClient.prompt()
            .user(message)
            .stream()
            .content();
    }

    // WebFlux ilə streaming
    @GetMapping("/chat")
    public Mono<String> chat(@RequestParam String message) {
        return Mono.fromCallable(() ->
            chatClient.prompt()
                .user(message)
                .call()
                .content()
        );
    }
}

// ─── Chat History — Conversation memory ──────────────────
@Service
public class ConversationService {

    private final ChatClient chatClient;
    private final Map<String, List<Message>> conversations = new ConcurrentHashMap<>();

    public String chat(String sessionId, String userMessage) {
        // Köhnə söhbəti yüklə
        List<Message> history = conversations.getOrDefault(sessionId, new ArrayList<>());

        // Yeni mesaj əlavə et
        history.add(new UserMessage(userMessage));

        // AI-yə göndər
        ChatResponse response = chatClient.prompt()
            .messages(history)
            .call()
            .chatResponse();

        // AI cavabını historia əlavə et
        AssistantMessage assistantMessage = response.getResult().getOutput();
        history.add(assistantMessage);

        // Saxla (production: Redis/DB)
        conversations.put(sessionId, history);

        return assistantMessage.getContent();
    }
}
```

```java
@Service
public class EmailGeneratorService {

    private final ChatClient chatClient;

    // Template faylından
    public String generateEmail(String customerName,
                                 String orderNumber,
                                 String issue) {
        PromptTemplate template = new PromptTemplate("""
            Müştəriyə cavab e-poçtu yaz.
            
            Müştəri adı: {customerName}
            Sifariş nömrəsi: {orderNumber}
            Problem: {issue}

            """);

        Prompt prompt = template.create(Map.of(
            "customerName", customerName,
            "orderNumber", orderNumber,
            "issue", issue
        ));

        return chatClient.prompt(prompt).call().content();
    }

    // Resource-dan template
    @Value("classpath:prompts/email-template.st")
    private Resource emailTemplate;

    public String generateFromFile(Map<String, Object> vars) {
        PromptTemplate template = new PromptTemplate(emailTemplate);
        return chatClient.prompt(template.create(vars)).call().content();
    }
}
```

```java
// ─── Custom output converter ──────────────────────────────
@Service
public class DataExtractionService {

    private final ChatClient chatClient;

    public record ExtractedData(
        String companyName,
        String email,
        String phone,
        String address
    ) {}

    public ExtractedData extractContactInfo(String rawText) {
        BeanOutputConverter<ExtractedData> converter =
            new BeanOutputConverter<>(ExtractedData.class);

        String prompt = """
            Aşağıdakı mətndən əlaqə məlumatlarını çıxar.
            
            Mətn: %s
            
            %s
            """.formatted(rawText, converter.getFormat()); // JSON schema əlavə edir

        String response = chatClient.prompt()
            .user(prompt)
            .call()
            .content();

        return converter.convert(response);
    }
}

// ─── MapOutputConverter ───────────────────────────────────
public Map<String, Object> analyzeText(String text) {
    MapOutputConverter converter = new MapOutputConverter();

    String response = chatClient.prompt()
        .user(text + "\n\n" + converter.getFormat())
        .call()
        .content();

    return converter.convert(response);
}
```

## RAG — Retrieval Augmented Generation

```java
// ─── RAG — Öz məlumatlarınla AI ──────────────────────────
// Problem: LLM 2024-ci ilədək bilir, şirkət məlumatını bilmir
// Həll: Şirkət sənədlərini embedding ilə axtarıb kontekst kimi ver

// ─── 1. Document ingestion ───────────────────────────────
@Service
public class DocumentIngestionService {

    private final VectorStore vectorStore;
    private final TokenTextSplitter textSplitter;

    public void ingestPdf(String filePath) {
        // PDF oxu
        PagePdfDocumentReader pdfReader = new PagePdfDocumentReader(
            new FileSystemResource(filePath));
        List<Document> documents = pdfReader.get();

        // Parçalara böl
        List<Document> chunks = textSplitter.apply(documents);

        // Vector store-a yaz (embedding avtomatik hesablanır)
        vectorStore.add(chunks);

        log.info("{} sənəd {} parçaya bölünüb ingested edildi", filePath, chunks.size());
    }

    public void ingestText(String content, Map<String, Object> metadata) {
        Document doc = new Document(content, metadata);
        List<Document> chunks = textSplitter.apply(List.of(doc));
        vectorStore.add(chunks);
    }
}

// ─── 2. RAG Query ────────────────────────────────────────
@Service
public class RagQueryService {

    private final ChatClient chatClient;
    private final VectorStore vectorStore;

    public String askWithContext(String question) {
        // Semantic search — oxşar sənədləri tap
        List<Document> relevantDocs = vectorStore.similaritySearch(
            SearchRequest.query(question)
                .withTopK(5)               // Ən yaxın 5 sənəd
                .withSimilarityThreshold(0.7) // Minimum oxşarlıq
        );

        // Sənədləri kontekst kimi birləşdir
        String context = relevantDocs.stream()
            .map(Document::getContent)
            .collect(Collectors.joining("\n\n---\n\n"));

        // LLM-ə kontekst + sual göndər
        return chatClient.prompt()
            .system("""
                Aşağıdakı kontekst əsasında sualı cavablandır.
                Kontekstdən kənar məlumat vermə.
                Cavabı bilmirsənsə "Bu haqda məlumatım yoxdur" de.
                """)
            .user("""
                Kontekst:
                %s
                
                Sual: %s
                """.formatted(context, question))
            .call()
            .content();
    }
}

// ─── QuestionAnswerAdvisor — Spring AI RAG helper ────────
@Configuration
public class RagConfig {

    @Bean
    public ChatClient ragChatClient(ChatClient.Builder builder,
                                     VectorStore vectorStore) {
        return builder
            .defaultAdvisors(
                // Built-in RAG advisor
                new QuestionAnswerAdvisor(vectorStore,
                    SearchRequest.defaults()
                        .withTopK(4)
                        .withSimilarityThreshold(0.75))
            )
            .build();
    }
}
```

```java
            @ToolParam(description = "Gün sayı (1-7)") int days) {

    }

// ─── Order management tools ───────────────────────────────
@Service
public class OrderManagementTools {

    private final OrderRepository orderRepository;

    @Tool(description = "Sifariş statusunu sifariş nömrəsinə görə yoxla")
    public String checkOrderStatus(
            @ToolParam(description = "Sifariş nömrəsi") String orderId) {
        return orderRepository.findById(orderId)
            .map(o -> "Sifariş %s statusu: %s".formatted(orderId, o.getStatus()))
            .orElse("Sifariş tapılmadı: " + orderId);
    }

    @Tool(description = "Sifarişi ləğv et")
    public String cancelOrder(
            @ToolParam(description = "Ləğv ediləcək sifariş nömrəsi") String orderId) {
        try {
            orderService.cancel(orderId);
            return "Sifariş " + orderId + " uğurla ləğv edildi";
        } catch (Exception e) {
            return "Sifariş ləğv edilə bilmədi: " + e.getMessage();
        }
    }
}
```

## İntervyu Sualları

### 1. Spring AI nədir?
**Cavab:** Spring ekosistemi üçün AI/LLM inteqrasiya framework-ü (2024). OpenAI, Anthropic, Google Gemini, Ollama (local) kimi çoxlu provider-i vahid API ilə idarə edir. Əsas xüsusiyyətlər: `ChatClient` (fluent API), Prompt Templates, Structured Output (POJO cavab), RAG (Retrieval Augmented Generation), Function Calling (AI Java metodlarını çağırır), Vector Store inteqrasiyası (PGVector, Redis, Pinecone). Provider dəyişikliyi minimal kod dəyişikliyi ilə olur.

### 2. RAG nədir?
**Cavab:** Retrieval Augmented Generation — LLM-i öz məlumatlarınla gücləndirmə. Problem: LLM training cutoff tarixi var, şirkət daxili məlumatları bilmir. RAG həlli: (1) Şirkət sənədlərini parçalara böl; (2) Hər parçanı embedding (vector) kimi Vector Store-a yaz; (3) Sual gəldikdə: sualı embedding-ə çevir → Vector Store-dan ən oxşar N parçanı tap → LLM-ə kontekst + sual göndər → Cavab. Spring AI-da `VectorStore.similaritySearch()` + `QuestionAnswerAdvisor` ilə asanlaşdırılır.

### 3. Function Calling nədir?
**Cavab:** AI model-in Java metodlarını çağırma qabiliyyəti. `@Tool` annotation ilə metod AI-ə açılır. User "sifariş 42-nin statusu?" dedikdə: AI `checkOrderStatus("42")` çağırır → Java metodu real DB-dən data alır → AI natural language ilə cavab verir. Spring AI-da `chatClient.prompt().tools(myService).call()` ilə qeydiyyat. Mürəkkəb iş axışları: AI birdən çox tool-u ardıcıl ya paralel çağıra bilər (agentic patterns).

### 4. Structured Output necə işləyir?
**Cavab:** `chatClient.prompt().user(prompt).call().entity(MyRecord.class)` — AI cavabını avtomatik JSON-a, sonra POJO-ya çevirir. Spring AI `BeanOutputConverter` ilə: Java class-dan JSON schema yaranır, prompt-a əlavə edilir, AI bu formatda cavab verir, cavab deserialize olunur. `List<String>` kimi generic tip üçün `ParameterizedTypeReference` istifadə edilir. Hata paylanmasını azaldır, tip güvənliyi sağlayır.

### 5. Streaming response niyə lazımdır?
**Cavab:** LLM cavabı token-by-token yaradır (bir-iki saniyəlik gecikməyib). Streaming olmadan: bütün cavab hazırlanana qədər client gözləyir (5-30 saniyə). Streaming ilə: hər token gəldikcə client görür — daha yaxşı UX. Spring AI `chatClient.prompt().stream().content()` — `Flux<String>` qaytarır. Server-Sent Events (SSE) ilə REST endpoint-dən stream: `MediaType.TEXT_EVENT_STREAM_VALUE`. WebFlux ilə reaktif integration mümkün.
